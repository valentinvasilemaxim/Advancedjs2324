            class Shape {
                color;
                setColor(color){
                    this.color = color;
                }
                getColor() {
                    return this.color;
                }
            }
    
            class Rectangle extends Shape {
                constructor( width, height) {
                    this.width = width;
                    this.height = height;
                }
    
                getArea() {
                    return this.width * this.height;
                }
            }
    
            let rectangle = new Rectangle( 5, 8);
            console.log("Area:" + rectangle.getArea());
