3.Ce afiseaza urmatorul code? Implicit Binding in JavaScript

        function mesaj() {
            console.log(this.name + ' are '+ this.cantitate+' mere');
        }

        const myObj = {
            cantitate: 22,
            name: 'Ana',
            mesaj:mesaj,
            newObj: {
                cantitate: 26,
                name: 'Ionel',
                mesaj:mesaj,
            }
        }

        myObj.mesaj(); // `this` is bound to `myObj` 
        myObj.newObj.mesaj(); // `this` is bound to `newObj`

8.Ce afiseaza urmatorul code? bind in JavaScript

        let masina = {
            name: 'Fiat',
            run: function (speed) {
                console.log(this.name + ' merge cu ' + speed + ' km/h.');
            }
        };
        let avion = {
            name: 'AirCar',
            fly: function (speed) {
                console.log(this.name + ' zboare cu ' + speed + ' km/h.');
            }
        };
        
        avion.fly(190);
        let run = masina.run.bind(avion, 160);
        run();
